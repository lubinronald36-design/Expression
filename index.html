const lyrics = [ { time: 0, text: "Your first lyric line" }, { time: 5, text: "Your second lyric line" }, { time: 10, text: "Your third lyric line" }, // Add more lyrics here... ]; // Get the audio and lyrics elements const audio = document.getElementById('audio'); const lyricsDiv = document.getElementById('lyrics'); // Function to update lyrics audio.addEventListener('timeupdate', () => { const currentTime = audio.currentTime; const currentLyric = lyrics.find((lyric) => currentTime >= lyric.time && (!lyric.endTime || currentTime < lyric.endTime)); if (currentLyric) { lyricsDiv.innerText = currentLyric.text; } });
